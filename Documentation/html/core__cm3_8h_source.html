895static __INLINE void NVIC_SetPriorityGrouping(uint32_t PriorityGroup)
896{
897 uint32_t reg_value;
898 uint32_t PriorityGroupTmp = (PriorityGroup & 0x07); /* only values 0..7 are used */
899
900 reg_value = SCB->AIRCR; /* read old register configuration */
901 reg_value &= ~(SCB_AIRCR_VECTKEY_Msk | SCB_AIRCR_PRIGROUP_Msk); /* clear bits to change */
902 reg_value = (reg_value |
903 (0x5FA << SCB_AIRCR_VECTKEY_Pos) |
904 (PriorityGroupTmp << 8)); /* Insert write key and priorty group */
905 SCB->AIRCR = reg_value;
906}

1058static __INLINE uint32_t NVIC_EncodePriority (uint32_t PriorityGroup, uint32_t PreemptPriority, uint32_t SubPriority)
1059{
1060 uint32_t PriorityGroupTmp = (PriorityGroup & 0x07); /* only values 0..7 are used */
1061 uint32_t PreemptPriorityBits;
1062 uint32_t SubPriorityBits;
1063
1064 PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1065 SubPriorityBits = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1066
1067 return (
1068 ((PreemptPriority & ((1 << (PreemptPriorityBits)) - 1)) << SubPriorityBits) |
1069 ((SubPriority & ((1 << (SubPriorityBits )) - 1)))
1070 );
1071}

1089{

1091 uint32_t PreemptPriorityBits;
1092 uint32_t SubPriorityBits;
1093
1094 PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
1095 SubPriorityBits = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
1096
1097 *pPreemptPriority = (Priority >> SubPriorityBits) & ((1 << (PreemptPriorityBits)) - 1);
1098 *pSubPriority = (Priority ) & ((1 << (SubPriorityBits )) - 1);
1099}
