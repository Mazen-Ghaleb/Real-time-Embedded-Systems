196 * Description :
197 * it used to control the light of the LED according to the temperature
198 */
199void threshold_LED(unsigned int ReadTemperature){
200 // Setpoint threshold
201 if (ReadTemperature > threshold) {
202 GPIOPinWrite(GPIO_PORTF_BASE,GPIO_PIN_1 | GPIO_PIN_2 | GPIO_PIN_3, GPIO_PIN_1); // Turns Red LED and turn off Heater
203 }
204 else if (ReadTemperature == threshold) {
205 GPIOPinWrite(GPIO_PORTF_BASE,GPIO_PIN_1 | GPIO_PIN_2 | GPIO_PIN_3, GPIO_PIN_1 | GPIO_PIN_3); // Turns Yellow LED
206 }
207 else if (ReadTemperature < threshold) {
208 GPIOPinWrite(GPIO_PORTF_BASE,GPIO_PIN_1 | GPIO_PIN_2 | GPIO_PIN_3, GPIO_PIN_2); // Turns Blue LED
209 }
210
211 //Alarm threshold
212 if (ReadTemperature > alarmThreshold) {
213 // Turn on Buzzard alarm
214 sprintf (buffer, "%s","The temperature is higher than alarm threshold, Turn Buzzard ON");
215 UART_PrintBuffer();
216 }
217}
