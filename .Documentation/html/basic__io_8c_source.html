41/* The ITM port is used to direct the printf() output to the serial window in
42the Keil simulator IDE. */
43#define mainITM_Port8(n) (*((volatile unsigned char *)(0xE0000000+4*n)))
44#define mainITM_Port32(n) (*((volatile unsigned long *)(0xE0000000+4*n)))
45#define mainDEMCR (*((volatile unsigned long *)(0xE000EDFC)))
46#define mainTRCENA 0x01000000

48void vPrintString( const char *pcString )
49{
50 /* Print the string, suspending the scheduler as method of mutual
51 exclusion. */
52 vTaskSuspendAll();
53 {
54 printf( pcString );
55 }
56 xTaskResumeAll();
57}

60void vPrintStringAndNumber( const char *pcString, unsigned long ulValue )
61{
62 /* Print the string, suspending the scheduler as method of mutual
63 exclusion. */
64 vTaskSuspendAll();
65 {
66 printf( "%s %u\n", pcString, ulValue );
67 }
68 xTaskResumeAll();
69}

72int fputc( int iChar, FILE *pxNotUsed )
73{
74 /* Just to avoid compiler warnings. */
75 ( void ) pxNotUsed;
76
77 if( mainDEMCR & mainTRCENA )
78 {
79 while( mainITM_Port32( 0 ) == 0 );
80 mainITM_Port8( 0 ) = iChar;
81 }
82
83 return( iChar );
84}
